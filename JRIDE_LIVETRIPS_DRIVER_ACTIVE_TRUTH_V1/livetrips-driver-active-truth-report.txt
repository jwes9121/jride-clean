with target_names(name) as (

    ('Warren Tuyapon II B.'),

), profile_hits as (
  select
    tn.name as searched_name,
    dp.driver_id,
    dp.full_name,
    dp.phone,
    dp.municipality
  from target_names tn
  left join driver_profiles dp
    on lower(coalesce(dp.full_name,'')) like '%' || lower(tn.name) || '%'
    or lower(tn.name) like '%' || lower(coalesce(dp.full_name,'')) || '%'
), driver_hits as (
  select
    ph.searched_name,
    ph.driver_id as profile_driver_id,
    ph.full_name,
    ph.phone,
    ph.municipality,
    d.id as drivers_id,
    d.driver_name,
    d.driver_status,
    d.wallet_locked,
    d.wallet_balance,
    d.min_wallet_required
  from profile_hits ph
  left join drivers d on d.id = ph.driver_id
), latest_locations as (
  select distinct on (driver_id)
    driver_id,
    status as location_status,
    lat,
    lng,
    town,
    home_town,
    updated_at
  from driver_locations
  order by driver_id, updated_at desc nulls last
)
select
  dh.*,
  ll.location_status,
  ll.town as location_town,
  ll.home_town,
  ll.updated_at as location_updated_at,
  case
    when lower(coalesce(dh.driver_status,'')) = 'deactivated' then 'BLOCK_BY_DRIVER_STATUS_DEACTIVATED'
    when lower(coalesce(dh.driver_status,'')) in ('inactive','disabled','suspended','blocked') then 'SHOULD_BLOCK_BY_DRIVER_STATUS_ALIAS'
    else 'NOT_BLOCKED_BY_CURRENT_LIVETRIPS_DRIVER_STATUS_RULE'
  end as current_livetrips_filter_result
from driver_hits dh
left join latest_locations ll on ll.driver_id = dh.profile_driver_id
order by dh.searched_name;
